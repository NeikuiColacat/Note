#include <iostream>
using namespace std;
//ax + by = gcd          
int exgcd(int a, int b, int& x, int& y) {
	if (!b) {
		x = 1, y = 0;
		return a;
	}
	int d =  exgcd(b, a % b, y, x);
	y = y - x * (a / b);
	return d;
}
signed main() {
	ios::sync_with_stdio(0);
	cin.tie(0);
	cout.tie(0);
	int n;
	cin >> n;
	while (n--) {
		int a, b;
		cin >> a >> b;
		int x ,y;
		exgcd(a,b,x,y);
		cout << x << ' ' << y<<endl;
	}

}

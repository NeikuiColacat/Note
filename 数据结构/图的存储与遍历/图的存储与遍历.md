## 用邻接表的方式来存储图
## 每个节点都用一个链表来存储它能走到的下一个节点
```cpp
const int N = 1e6;
int head[N];
int ne[N];
int e[N];
int idx;
signed main(){
    memset(head,-1,sizeof(head))
}
```

## 遍历  深度优先
```cpp
int dfs(int u)
{
    st[u] = true; // st[u] 表示点u已经被遍历过

    for (int i = h[u]; i != -1; i = ne[i])
    {
        int j = e[i];
        if (!st[j]) dfs(j);
    }
}
```

## 宽度优先
```cpp
queue<int> q;
st[1] = true; // 表示1号点已经被遍历过
q.push(1);

while (q.size())
{
    int t = q.front();
    q.pop();

    for (int i = h[t]; i != -1; i = ne[i])
    {
        int j = e[i];
        if (!st[j])
        {
            st[j] = true; // 表示点j已经被遍历过
            q.push(j);
        }
    }
}
```

#include<bits/stdc++.h>
using namespace std;
#define int long long
const int N = 1e3+10;
const int M = 2*1e6 + 10;
int arr[N][N] = {0};
struct coordinate{
	int x;
	int y;
};
coordinate P1[M];
coordinate P2[M];
signed main() {
	//读入数据
	int n, m, q;
	cin >> n >> m >> q;
	for (int i = 1; i <= n; i++) {
		for (int j = 1; j <= m; j++) cin >> arr[i][j];
	}
	for (int i = 1; i <= q; i++)cin >> P1[i].x >> P1[i].y >> P2[i].x >> P2[i].y;
	//-----------------------------------------------------------------------------
	//求矩阵前缀和
	for (int i = 1; i <= n; i++) {
		for (int j = 1; j <= m; j++) {
			arr[i][j] = arr[i - 1][j] + arr[i][j - 1] - arr[i - 1][j - 1] + arr[i][j];
		}
	}
	//求子矩阵和
	int ans = 0;
	for (int i = 1; i <= q; i++) {
		ans = arr[P2[i].x][P2[i].y] - arr[P1[i].x - 1][P2[i].y] - arr[P2[i].x][P1[i].y - 1] + arr[P1[i].x - 1][P1[i].y - 1];
		cout <<ans<< endl;
	}
	
	
	return 0;
}

# $C_{2n}^n-C_{2n}^{n-1}=\frac{C_{2n}^n}{n+1}$

```cpp
#include<iostream>
using namespace std;
const int p = 1e9+7;
typedef long long ll;
int fp(int a,int b){
    int res = 1;
    while(b){
        if(b&1)res=(ll)res*a%p;
        a = (ll)a*a%p;
        b>>=1;
    }
    return res;
}
int main(){
    int n;
    cin>>n;
    int a=2*n,b=n;
    int res =1;
    for(int i = a,j=1;j<=b;i--,j++){
        res = (ll)res*i%p;
        res = (ll)res*fp(j,p-2)%p;
    }
    res =(ll)res*fp(n+1,p-2)%p;
    cout<<res;
    return 0;
}
```
